\section*{Proofs}
\begin{theorem}
$0v=0$ for all $v \in V$
\end{theorem}
\begin{proof}
$\forall v \in V$, 
\begin{align*}
0v &= (0 + 0)v\\
&= 0v + 0v\\
-0v + 0v &= -0v + (0v + 0v)\\
0 &= (-0v + 0v) + 0v \\
0 &= 0v
\end{align*}
\end{proof}
